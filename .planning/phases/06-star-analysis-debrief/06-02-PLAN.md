```typescript
export async function generateDebrief(entries: TranscriptEntry[]): Promise<DebriefReport> {
  const apiKey = process.env.NEXT_PUBLIC_GEMINI_API_KEY;
  if (!apiKey) throw new Error("API Key missing");

  // Step 1: Process transcript
  const { merged, pairs, stats } = processTranscript(entries);

  // Step 2: Format for AI
  const transcriptText = formatTranscriptForAI(merged);
  const prompt = buildAnalysisPrompt(transcriptText, pairs);

  // Step 3: Call Gemini
  const genAI = new GoogleGenerativeAI(apiKey);
  const model = genAI.getGenerativeModel({ model: "gemini-2.0-flash" });

  try {
    const result = await model.generateContent(prompt);
    const text = result.response.text();
    const jsonMatch = text.match(/\{[\s\S]*\}/);

    if (!jsonMatch) throw new Error("Failed to parse JSON");

    const aiOutput = JSON.parse(jsonMatch[0]);

    // Step 4: Validate patterns (D-03: filter out patterns with < 3 instances)
    const validPatterns = (aiOutput.analysis?.patterns || []).filter(
      (p: Pattern) => p.instanceCount >= 3
    );

    // Step 5: Build report
    return {
      // Legacy fields (derive from analysis or use defaults)
      elevatorPitch: deriveElevatorPitch(aiOutput),
      keyAchievements: aiOutput.analysis?.strengths?.slice(0, 3) || [],
      uniqueValueProposition: aiOutput.analysis?.strengths?.[0] || "",
      areasForImprovement: aiOutput.analysis?.weaknesses || [],

      // Phase 4 fields
      transcriptSummary: buildTranscriptSummary(pairs),
      sessionStats: stats,

      // Phase 6 fields
      evaluations: aiOutput.evaluations || [],
      analysis: {
        ...aiOutput.analysis,
        patterns: validPatterns,
      },
      coaching: aiOutput.coaching || buildFallbackCoaching(),
    };
  } catch (error) {
    console.error("Analysis generation failed", error);
    return buildFallbackReport(pairs, stats);
  }
}
```
