    ```typescript

    ```

    5. Add extraction function:
    ```typescript
    const extractPersonaData = async () => {
      if (!interviewData.resume.trim() || !interviewData.jobDescription.trim()) {
        return;
      }
      setPersonaLoading(true);
      setPersonaError(null);
      try {
        const response = await fetch('/api/extract-persona', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            resume: interviewData.resume,
            jobDescription: interviewData.jobDescription,
          }),
        });
        if (!response.ok) throw new Error('Extraction failed');
        const persona = await response.json();
        setInterviewData(prev => ({ ...prev, candidatePersona: persona }));
      } catch (e) {
        setPersonaError(e instanceof Error ? e.message : 'Failed to extract persona');
      } finally {
        setPersonaLoading(false);
      }
    };
    ```

    6. Add handler for persona changes:
    ```typescript
    const handlePersonaChange = (persona: CandidatePersona) => {
      setInterviewData(prev => ({ ...prev, candidatePersona: persona }));
    };
    ```

    7. Update handleStartInterview to navigate to 'persona' step instead of 'interview':
    ```typescript
    const handleStartInterview = (duration: number) => {
      // ... existing logic ...
      extractPersonaData(); // Trigger extraction
      setStep('persona'); // Go to persona screen first
    };
    ```

    8. Add handleProceedToSimulation:
    ```typescript
    const handleProceedToSimulation = () => {
      // For now, go to interview step
      // Phase 8 will add actual simulation logic

    };
    ```

    ```typescript
        onProceed={handleProceedToSimulation}
        onBack={() => setStep('setup')}
      />
    )}
    ```
